// lib/qrPdfGenerator.ts
import jsPDF from 'jspdf'
import QRCode from 'qrcode'

interface QRLabelConfig {
  format: 'a4-3x7' | 'a4-5x10' | 'single'
  codes: string[]
}

export async function generateQRLabelsPDF(config: QRLabelConfig): Promise<Blob> {
  const pdf = new jsPDF('p', 'mm', 'a4')

  const layouts = {
    'a4-3x7': { cols: 3, rows: 7, width: 60, height: 35, margin: 10 },
    'a4-5x10': { cols: 5, rows: 10, width: 38, height: 27, margin: 5 },
    'single': { cols: 1, rows: 1, width: 62, height: 30, margin: 0 }
  }

  const layout = layouts[config.format]
  let page = 0
  let col = 0
  let row = 0

  for (const code of config.codes) {
    if (row >= layout.rows) {
      pdf.addPage()
      page++
      row = 0
    }

    const x = layout.margin + col * (layout.width + 5)
    const y = layout.margin + row * (layout.height + 5)

    // Generiraj QR kodo kot sliko
    const qrDataUrl = await QRCode.toDataURL(code, {
      width: layout.width * 2,
      margin: 1
    })

    // Dodaj QR sliko
    pdf.addImage(qrDataUrl, 'PNG', x, y, layout.width - 10, layout.width - 10)

    // Dodaj kodo pod QR
    pdf.setFontSize(10)
    pdf.text(code, x + layout.width / 2, y + layout.height - 3, { align: 'center' })

    col++
    if (col >= layout.cols) {
      col = 0
      row++
    }
  }

  return pdf.output('blob')
}
